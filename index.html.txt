        function renderStars() {
            $('.rating').each(function() {
                $(this).empty();
                var rating = ($(this).attr('data-rating') * 2).toFixed() / 2;
                for (var stars = 1; stars <= rating; stars++) {
                    $(this).append('<span class="glyphicon glyphicon-star" aria-hidden="true"></span>');
                    if (rating - stars == 0.5) {
                        $(this).append('<span class="glyphicon glyphicon-star" aria-hidden="true" style="width: 7px; overflow: hidden;"></span>');
                    }
                }
            });
        }
